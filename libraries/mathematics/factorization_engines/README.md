```c
void inflate(struct pure_square *square_pair) {
    square_pair->square += square_pair->root;
    square_pair->root += 1;
    square_pair->square += square_pair->root;
}
```

```c
while (pair_one->square != pair_two->square) {

	inflate(pair_one);
    while (pair_one->square > pair_two->square)
	inflate(pair_two);
}

struct pure_square factors =
    { pair_one->root - pair_two->root,
    pair_one->root + pair_two->root };
```
